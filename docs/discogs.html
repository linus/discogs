Make a request

 
  discogsRequest = (url, next) ->
    rparse = (err, res, body) ->
      if err
        return next err
      if res.headers['content-type']?.indexOf('json') >= 0 or not format
        try
          body = JSON.parse(body)
        catch e
          err = e
          body = null
      next err, body

    request
      uri: getUrl url
      headers: {'accept-encoding': 'gzip'}
      encoding: null
      proxy: sm.app.Env.getDefault().getHTTPProxy()
      (error, res, body) =>
        if not error and 200 <= res.statusCode < 400 and body
          if res.headers['content-encoding']?.indexOf('gzip') >= 0
            zlib.gunzip body, (err, body) =>
              rparse err, res, body
          else
            rparse null, res, body
          return
        next error

  responseHandler = (type, next) ->
    (err, res) ->
      return next(err, res) if err or res not instanceof Object or type not of res?.resp
      next(null, res.resp[type])
